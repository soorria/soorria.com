```vue title="Vue 2.7/3 (Composition API)"
  <div>{{ message }}</div>
</template>

<script lang="ts">
import { defineComponent, ref } from 'vue'

  setup() {
    const message = ref('Hello, World!')

    return {
      message,
    }
  },
})
</script>
```

```vue title="Vue 2/3 (Options API)"
<template>
  <div>{{ message }}</div>
</template>

<script lang="ts">
import { defineComponent, ref } from 'vue'

  data() {
    return {
      message: 'Hello, World!',
    }
  },
})
</script>
```
