title: debounce
shortDescription: Run a function at most once every N seconds, at the end of the N seconds.

See [throttle](/snippets/throttle) if you want your function to run at the start of the N seconds.

```typescript
const debounce = <T extends CallableFunction>(fn: T, delay: number): T => {
  let timer: ReturnType<typeof setTimeout>

  return ((...args: any[]) => {
    if (timer) clearTimeout(timer)
    setTimeout(() => {
      fn(...args)
    }, delay)
  }) as any
}
```
