---
title: throttle
short_description: Run a function at most once every N seconds, at the start of the N seconds.
category: TypeScript
tags:
  - typescript
created_at: '2021-03-28'
updated_at: '2021-03-28'
---

See [debounce](/snippets/debounce) if you want your function to run at the end of the N seconds.

```typescript
const throttle = <T extends CallableFunction>(fn: T, delay: number): T => {
  let canRun = true

  return ((...args: any[]) => {
    if (canRun) {
      fn(...args)
      canRun = false
      setTimeout(() => {
        canRun = true
      }, delay)
    }
  }) as any
}
```
